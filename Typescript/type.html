// 需求 传入什么值 返回什么值 
function id1<Type>(value: Type): Type {
    return value
}

let res1 = id1<string>('a')

  1. 泛型函数 function id1<Type>(value: Type): Type { return value } 其中 Type 是一种特殊的类型变量，可以是其他值（合法的类型名称）
// 需求 传入什么值 返回什么值 
function id1<Type>(value: Type): Type {
    return value
}
let res = id1<number>(1)
let res1 = id1<string>('a')
let res2 = id1<boolean>(true)
  1. 泛型约束 （上面的 Type可以是任何类型 ） 有时候 需要为泛型添加约束条件，缩窄泛型的取值范围 1指定更加具体的类型 2 添加约束 extends
// 需求 传入什么值 返回什么值  访问 length属性
function id1<Type>(value: Type): Type {
    console.log(value.length) // 报错  不是所有的类型都有length 属性的 
    return value
}
function id2<Type>(value: Type[]): Type[] {
    console.log(value.length) // 报错  不是所有的类型都有length 属性的 
    return value
}

interface Ilength {
    length:number
}
//   Type 要满足 Ilength 的约束
function id3<Type extends Ilength>(value: Type): Type {
    console.log(value.length)
    return  value
}
id3([1])
id3({length:1111})
id3('111111111111111111')
  1. 泛型接口： 在使用泛型接口的时候 需要指定具体的类型 obj:Idrunc<string>
interface Idrunc<Type>{
    id:(value:Type) => Type  // 
    ids:() => Type[]  // 
}

let obj:Idrunc<string> = {
    id:(value:string) => {
        return '123123'
    },
    ids:() => {
        return ['1']
    }
}

  1. 泛型工具类型
  • Partial 可选属性 创建一个新的类型 不会改变原类型
  • Readonly 所有属性都变为只读属性
  • Pick 从Type中选择一组属性构造新的类型
  • Record
 interface Props {
    a:number,
    b:string,
    c:number[]
}
type MyProps = Partial<Props> //都加问号
type MyProps1 = Readonly<Props> //都加Readonly
type MyProps2 = Pick<Props,'a'|'b';> //  只有a和b两个属性



type Atype = Record<'aa'|'bb'|'cc',string[]>

type Btype = {
    aa: string[];
    bb: string[];
    cc: string[];
}
