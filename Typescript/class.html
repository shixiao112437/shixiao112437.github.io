interface aa {
    say(num: number): string
}
class Person implements aa {
    // 添加属性
    age: number
    gerder: string

    constructor(age, gerder) {
        this.age = age
        this.gerder = gerder
    }
    say: (num: number) => string = () => {
        return '111'
    }

}
let people = new Person(11, 'nan')

# 类成员可见性

  1. 可见性修饰符public公有的（默认值）、protected受保护的(仅在其声明的类和子类中存在、不包括实例对象)、private私有的（只在当前类中可见，实例及子类都不可见）。

    • readonly 表示只读，用来防止在构造函数（constructor）之外对属性进行赋值,只能在属性前。
    • 接口或者对象类型也可以使用readonly
class Animal {
    readonly age: number = 18 //  只读 默认值
    readonly age1 = 18 //  只读  字面量类型 只能是18

    constructor(age:number){
        this.age = age
        this.age1 = 17 // 报错不能将类型17分配给类型18

    }
    
    /* 所有的都可以访问 */
    public say(){
        console.log('say')
    }
    /* 实例的对象是不可以访问的 */
    protected say1(){
        console.log('say')
    }
    /* 仅对当前类存在 */
    private say2(){
        console.log('say')
    }
}
interface Person {
    readonly name: string

}
let obj: {readonly name:string} = {name:'lll'}
let obj1: Person = {name:'111'}

obj.name = 'ssss' // 报错
obj1.name = 'ssss' // 报错
