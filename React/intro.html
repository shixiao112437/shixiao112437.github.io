# 权限路由封装(render方法)

// AuthRoute.jsx
import React, { Component } from 'react'
import {Route ,Redirect } from 'react-router-dom'
import { isAuth } from '../../utils/token'
// 在哪用 就在哪导
export default class AuthRoute extends Component {
  render() {
    console.log('AuthRoute接受参数props',this.props)
    let {exact, path ,Yemian } =this.props
    return <Route 
          // 直接写exact 就是exact={true}
          exact={exact} 
          path={path}//不要写死 传进来
          render={(props)=>{ //props组件的我们经常用的那个props 现在没什么用
            // 这种写法 就可以 在函数里面 做判断啦
            if( isAuth() ){ //登录了token 正常 显示对应页面 
              // 显示的页面 也不能写死
              // console.log('AuthRoute封装的render里面的props',props)
              // props 路由信息 {...props} 结构出 props里面的代码 原样传进去
              return  <Yemian {...props} ></Yemian>
            }else{ //没有登录 跳转到登录页面
                return <Redirect to="/login"></Redirect>
            }
          }}
        ></Route>
  }
}
// App.jsx
import React ,{ lazy,Suspense } from 'react'
import {BrowserRouter as Router,Route,Link ,Redirect} from 'react-router-dom'
import Rent  from './pages/Rent'
import Detail  from './pages/Detail'
import AuthRoute from './components/AuthRoute'

export default class App extends React.Component{
    render(){
        retrun {
            <Router>
                // 重定向
                  <Route 
                  exact 
                  path="/" 
                  render={(props)=>{ //props组件的我们经常用的那个props 现在没什么用
                    // 重定向  重新跳转 到首页 /home/index
                    return <Redirect to="/home/index"></Redirect>
                  }}
                ></Route>
                // 普通路由
            	<Route exact path='/rent' component={Rent}></Route>
            	// 鉴权路由
                // 
                <AuthRoute exact path='/detail' Yemian={Detail}>
                </AuthRoute>
            </Router>
        }
    }
}

# useReducer

  import react,{useReducer} from 'react'
  
  const initialState = {count: 0};
  function reducer(state, action) {
    switch (action.type) {
      case 'increment':
        return {count: state.count + 1};
      case 'decrement':
        return {count: state.count - 1};
      default:
        throw new Error();
    }
  }
  
  function App() {
      let [state,dispatch] = useReducer(reducer,initialState)
      // state 初始值为第二个参数initialState 
      // dispatch (action) 触发reducer修改state  同于 react-redux
      retrun (
      	<div>
          	Count: {state.count}
               <button onClick={() => dispatch({type: 'decrement'})}>-</button>
        			<button onClick={() => dispatch({type: 'increment'})}>+</button>
          </div>
      )
  }

# 回顾React,context的基本使用

  1. 在组件外面 调用 React.createContext() 得到Provider Consumer
  2. 用Provider 包裹住祖宗 写上value 数据
  3. 哪个孙子用 就是要 Consumer 包裹对应孙子
import React from 'react'
import ReactDOM from 'react-dom'
import './index.css'
// 组件通讯：
//祖孙传值 --不常用 不方便 
//1 从得到的对象里面 解构 拿出 Provider Consumer 来用
let {Provider,Consumer}=React.createContext()
// console.log('obj',obj) ;obj={Provider,Consumer}
class Zuzong extends React.Component {
  state={
    count:1,
    arr:[1,2,3]
  }
  render() {
      //2 用Provider 包裹住祖宗 写上value 数据  必须叫 value 只能写value
      // 任何数据都可以传
      return <Provider value={this.state.count}>
            <div>
              <h1>祖宗</h1>
              <Child  />
          </div>
      </Provider>
  }
}
class Child extends React.Component {
  render() {
      return (
         <div>
            <h1>Child 孩子</h1>
            <Sunzi></Sunzi>
         </div>
      )
  }
}
class Sunzi extends React.Component {

  render() {
      return (
          // 哪个孙子用 就是要 Consumer
          <Consumer>
              {(data)=>{ //data对应祖宗的value值数据
                 return <div>
                        <h1>孙子---接受的数据--{data}</h1>
                    </div> 
              }}
          </Consumer>
          
      )
  }
}

function WithGetSize (Com) {
    // 返回一个组件
    return    class extends Component{
        state= {
            x:document.documentElement.clientWidth,
            y:document.documentElement.clientHeight,
        }
        getSize = () =>{
            this.setState({
                x:document.documentElement.clientWidth,

        componentDidMount(){
            window.addEventListener('resize',this.getSize)

}
