# 基本使用

  • 安装： yarn add react-router-dom

# 路由传参

  • params参数
  <Link to={`/artic/detail/${item._id}`}> 查看</Link> // 传参
  <Route path='/artic/detail/:id' component={Detail}></Route> // 声明接受
//this.props.match.params.id // 组件内接受
  • search参数(query参数)
    <Link to={`/artic/detail/?id=${id}&title=${title}}`}> 查看</Link>
    let str = this.props.location.search  // 接受 str 为字符串 ?id=1&title=xxxx
    let {id,title} = qs.parse(str.slice(1))  // querystring
  • state参数
    <Link to={{
        pathname:'/artic/detail',
        state:{
            id:item._id,
            title:item.title
        }
    }}> 查看</Link>
    // 接受state参数
    const { id,title} = this.props.location.state
