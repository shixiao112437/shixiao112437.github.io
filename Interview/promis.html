# 实现promise

        const PENDING = 'PENDING'
        const FULFILED = 'FULFILED'
        const REJECTED = 'REJECTED'
        class Promiss1 {
            constructor(callback) {
                this.status = PENDING // promiss状态
                this.value = null // 解决值
                this.reason = null // 解决失败原因
                this.successCbs = [] //成功回调
                this.errorCbs  = [] // 失败回调

                const resolve = (val) => {
                    if(val instanceof this.constructor){ 
                        // val instanceof Promiss1 
                        // 说明resolve 的值还是一个promiss 实例
                      /*   val.then(val_ => {
                            resolve(val_)
                        }) // */
                        val.then(resolve,rejected) // 
                        return

                    }
                    if (this.status == PENDING) {
                        this.status = FULFILED
                        this.value = val
                        setTimeout(() => {
                            this.successCbs.forEach(cb => {
                                cb(this.value)
                            })
                        })
                    }
                }

                const rejected = (reason) => {
                    if (this.status == PENDING) {
                        this.status = REJECTED
                        this.reason = reason //错误原因
                        setTimeout(() => {
                            this.errorCbs.forEach(cb => {
                                cb(this.reason)
                            })
                        })
                    }
                }

                callback(resolve, rejected)
            }

            then(succCb, errCb) {
                if (this.status == FULFILED) {
                    // 异步执行成功回调
                    setTimeout(() => {
                        succCb(this.value)
                    })
                }else if(this.status == REJECTED){
                    setTimeout(() => {
                        errCb(this.reason)
                    })
                }
                if(this.status == PENDING){
                    this.successCbs.push(succCb)
                    this.errorCbs.push(errCb)
                }
            }
        }
